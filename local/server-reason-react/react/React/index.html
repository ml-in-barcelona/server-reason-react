type element =
  1. | Lower_case_element of {
    1. key : string option;
    2. tag : string;
    3. attributes : JSX.prop list;
    4. children : element list;
    }
  2. | Upper_case_component of unit -> element
  3. | Async_component of unit -> element Lwt.t
  4. | Client_component of {
    1. props : client_props;
    2. client : element;
    3. import_module : string;
    4. import_name : string;
    }
  5. | List of element array
  6. | Text of string
  7. | InnerHtml of string
  8. | Fragment of element

  10. | Provider of element
  11. | Consumer of element

    1. key : string option;

    3. fallback : element;
